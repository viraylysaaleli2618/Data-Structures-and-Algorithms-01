# Data-Structures-and-Algorithms-01
FOR LOOP, WHILE LOOP, AND DO-WHILE LOOP 
#include <iostream>
using namespace std;

int main() {
    int i = 0;

    // Print numbers 0 to 10
    while (i <= 10) {
        cout << i << " ";
        i++;
    }
    cout << endl;

    int j = 10;

    // Print numbers 10 to 0
    while (j >= 0) {
        cout << j << " ";
        j--;
    }
    cout << endl;

    return 0;
}
